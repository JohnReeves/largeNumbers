Using long long ints

copy & paste into yet another codepen

This code snippet outputs a list of factorials, as it gets past factorial(20) it makes a huge error!

#include <iostream>

using namespace std;

long long factorial(long long N) {
    long long temp;

    if (N <= 1) return 1;
    temp = N * factorial(N - 1);
    return temp;
}

int main()
{

    int t;
    long long int n,m;

    n=1234567890123456789;
    m=12345678901234567890;
    cout << n << m << endl;

    for(t=1; t <= 21; t++)
        cout << factorial(t) << endl;

    return 0;
}

Using double and setprecision 

#include <iostream>
#include <iomanip>

using namespace std;

double factorial(long long N) {
    double temp;

    if (N <= 1) return 1;
    temp = N * factorial(N - 1);
    return temp;
}

int main()
{

    int t;
    long long int n,m;

    n=1234567890123456789;
    m=12345678901234567890;
    cout << n << m << endl;

    for(t=1; t <= 30; t++)
        cout << setprecision(42) << factorial(t) << endl;

    return 0;
}
